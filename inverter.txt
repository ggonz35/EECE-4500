M1 (output in out) (input in 0) 0 pmosModel W=tpw L=tpl
M2 (output in 0) (input in 0) 0 nmosModel W=tnw L=tnl

Vdd vdd 0 DC 1.8V
Vgnd gnd 0 DC 0V

.model pmosModel PMOS LEVEL=54
.model nmosModel NMOS LEVEL=54

.param tplv = 1.0
.param tpwv = 1.0
.param tnln = 1.0
.param tnwn = 1.0

.tran 1ns 50ns

.control
    set tpl = tpl * tplv
    set tpw = tpw * tpwv
    set tnl = tnl * tnln
    set tnw = tnw * tnwn
.endc

.end
